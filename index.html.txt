    <script>
        // 简化版jQuery选择器
        const $ = (selector) => {
            const elements = document.querySelectorAll(selector);
            if (elements.length === 0) return null;
            if (elements.length === 1) return elements[0];
            return elements;
        };

        // 简化版事件绑定
        const on = (element, event, handler) => {
            if (element) {
                if (element.length) {
                    element.forEach(el => el.addEventListener(event, handler));
                } else {
                    element.addEventListener(event, handler);
                }
            }
        };

        // 1. 侧边栏导航切换
        on($('.nav-link'), 'click', function(e) {
            e.preventDefault();
            // 移除所有active样式
            $('.nav-link').forEach(link => link.classList.remove('active'));
            // 给当前点击的添加active
            this.classList.add('active');
            // 隐藏所有模块
            $('.module-section').forEach(section => section.classList.add('d-none'));
            // 显示对应模块
            const targetId = this.getAttribute('href');
            $(targetId).classList.remove('d-none');
        });

        // 2. 身份认证表单提交
        on($('#authForm'), 'submit', function(e) {
            e.preventDefault();
            const username = $('#username').value;
            const password = $('#password').value;
            const totpCode = $('#totpCode').value;
            
            if (!username || !password || !totpCode) {
                alert('账号、密码、动态验证码不能为空！');
                return;
            }
            
            // 模拟认证成功
            alert(`身份认证成功！\n账号：${username}\n已获取设备使用权限`);
            this.reset();
        });

        // 3. 获取动态验证码
        on($('#getTotpBtn'), 'click', function() {
            // 模拟生成6位验证码
            const totp = Math.floor(100000 + Math.random() * 900000);
            $('#totpCode').value = totp;
            alert(`动态验证码已生成：${totp}\n（实际场景需通过TOTP算法生成并发送）`);
        });

        // 4. 生物识别验证
        on($('#bioAuthBtn'), 'click', function() {
            const bioType = document.querySelector('input[name="bioType"]:checked');
            if (!bioType) {
                alert('请选择生物识别类型（人脸/指纹）！');
                return;
            }
            const typeName = bioType.value === 'face' ? '人脸' : '指纹';
            alert(`${typeName}验证成功！`);
        });

        // 5. 设备权限分配
        on($('#assignPermBtn'), 'click', function() {
            const projectGroup = $('#projectGroup').value;
            const user = $('#userSelect').value;
            const device = $('#deviceSelect').value;
            const useTime = $('#useTime').value;
            const operPerm = document.querySelector('input[name="operPerm"]:checked').value;
            
            if (!projectGroup || !user || !device || !useTime) {
                alert('请填写完整权限分配信息！');
                return;
            }

            // 模拟获取名称
            const projectName = $('#projectGroup option:checked').text;
            const userName = $('#userSelect option:checked').text;
            const deviceName = $('#deviceSelect option:checked').text;
            const permName = operPerm === 'read' ? '只读' : '读写';

            // 添加到表格
            const tr = document.createElement('tr');
            tr.innerHTML = `
                <td>${projectName}</td>
                <td>${userName}</td>
                <td>${deviceName}</td>
                <td>${useTime}</td>
                <td>${permName}</td>
                <td><button class="btn btn-sm btn-danger delete-perm">删除</button></td>
            `;
            $('#permTableBody').appendChild(tr);

            // 绑定删除事件
            on(tr.querySelector('.delete-perm'), 'click', function() {
                tr.remove();
            });

            alert('权限分配成功！');
            $('#resetPermBtn').click();
        });

        // 6. 重置权限分配表单
        on($('#resetPermBtn'), 'click', function() {
            $('#projectGroup').value = '';
            $('#userSelect').value = '';
            $('#deviceSelect').value = '';
            $('#useTime').value = '';
            document.querySelector('input[name="operPerm"][value="write"]').checked = true;
        });

        // 7. 日志查询
        on($('#searchLogBtn'), 'click', function() {
            const date = $('#logDate').value;
