    <script>
        // 1. 侧边栏导航切换
        $('.nav-link').click(function() {
            // 移除所有active样式
            $('.nav-link').removeClass('active');
            // 给当前点击的添加active
            $(this).addClass('active');
            // 隐藏所有模块
            $('.module-section').addClass('d-none');
            // 显示对应模块
            const targetId = $(this).attr('href');
            $(targetId).removeClass('d-none');
        });

        // 2. 身份认证表单提交
        $('#authForm').submit(function(e) {
            e.preventDefault();
            const username = $('#username').val();
            const password = $('#password').val();
            const totpCode = $('#totpCode').val();
            
            if (!username || !password || !totpCode) {
                alert('账号、密码、动态验证码不能为空！');
                return;
            }
            
            // 模拟认证成功
            alert(`身份认证成功！\n账号：${username}\n已获取设备使用权限`);
            $(this)[0].reset();
        });

        // 3. 获取动态验证码
        $('#getTotpBtn').click(function() {
            // 模拟生成6位验证码
            const totp = Math.floor(100000 + Math.random() * 900000);
            $('#totpCode').val(totp);
            alert(`动态验证码已生成：${totp}\n（实际场景需通过TOTP算法生成并发送）`);
        });

        // 4. 生物识别验证
        $('#bioAuthBtn').click(function() {
            const bioType = $('input[name="bioType"]:checked').val();
            if (!bioType) {
                alert('请选择生物识别类型（人脸/指纹）！');
                return;
            }
            const typeName = bioType === 'face' ? '人脸' : '指纹';
            alert(`${typeName}验证成功！`);
        });

        // 5. 设备权限分配
        $('#assignPermBtn').click(function() {
            const projectGroup = $('#projectGroup').val();
            const user = $('#userSelect').val();
            const device = $('#deviceSelect').val();
            const useTime = $('#useTime').val();
            const operPerm = $('input[name="operPerm"]:checked').val();
            
            if (!projectGroup || !user || !device || !useTime) {
                alert('请填写完整权限分配信息！');
                return;
            }

            // 模拟获取名称
            const projectName = $('#projectGroup option:selected').text();
            const userName = $('#userSelect option:selected').text();
            const deviceName = $('#deviceSelect option:selected').text();
            const permName = operPerm === 'read' ? '只读' : '读写';

            // 添加到表格
            const tr = `<tr>
                <td>${projectName}</td>
                <td>${userName}</td>
                <td>${deviceName}</td>
                <td>${useTime}</td>
                <td>${permName}</td>
                <td><button class="btn btn-sm btn-danger delete-perm">删除</button></td>
            </tr>`;
            $('#permTableBody').append(tr);

            // 绑定删除事件
            $('.delete-perm').click(function() {
                $(this).parents('tr').remove();
            });

            alert('权限分配成功！');
            $('#resetPermBtn').click();
        });

        // 6. 重置权限分配表单
        $('#resetPermBtn').click(function() {
            $('#projectGroup').val('');
            $('#userSelect').val('');
            $('#deviceSelect').val('');
            $('#useTime').val('');
            $('input[name="operPerm"][value="write"]').prop('checked', true);
        });

        // 7. 日志查询
        $('#searchLogBtn').click(function() {
            const date = $('#logDate').val();
            const type = $('#logType').val();
            const user = $('#logUser').val();
            alert(`日志查询条件：\n日期：${date || '全部'}\n类型：${type || '全部'}\n人员：${user || '全部'}\n（实际场景需对接后端接口查询）`);
        });

        // 8. 导出日志
        $('#exportLogBtn').click(function() {
            alert('日志已导出为Excel文件！（实际场景需实现文件导出逻辑）');
        });

        // 9. 应急授权表单提交
        $('#emergencyForm').submit(function(e) {
            e.preventDefault();
            const device = $('#emergencyDevice').val();
            const expire = $('#emergencyExpire').val();
            const reason = $('#emergencyReason').val();
            
            if (!device || !expire || !reason) {
                alert('请填写完整应急授权信息！');
                return;
            }

            // 生成随机授权码
            const authCode = Math.floor(100000 + Math.random() * 900000);
            const deviceName = $('#emergencyDevice option:selected').text();
            
            // 显示授权码区域
            $('#tempAuthCode').text(authCode);
            $('#authExpireTime').text(expire);
            $('#authDeviceName').text(deviceName);
            $('#authCodeArea').show();

            alert('临时授权码生成成功！');
        });

        // 10. 撤销授权码
        $('#revokeAuthBtn').click(function() {
            if (confirm('确定要撤销该授权码吗？')) {
                $('#authCodeArea').hide();
                alert('授权码已撤销！');
            }
        });
    </script>
